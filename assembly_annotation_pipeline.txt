### Blobtools

# Create BlobDir dataset
blobtools create
	--fasta /path/to/assembly/input.fasta \
	Dpini_blobtools_out

# Adding BLAST hits (taxonomic annotation for each sequence)
pb blastn -db nt \
	-query /path/to/assembly/input.fasta \
	-outfmt "6 qseqid staxids bitscore std" \
	-max_target_seqs 10 \
	-max_hsps 1 \
	-evalue 1e-25 \
	-num_threads $SLURM_CPUS_PER_TASK \
	-out BLAST.out

# download UniProtKB filtered for Hymenoptera https://www.uniprot.org/uniprotkb?query=%28taxonomy_id%3A7399%29 (accessed 15.3.2024)
# create Diamond DB 
diamond makedb --in UniProt_hymenoptera.faa -d UniProtHymDB
diamond blastx \
	--query /path/to/assembly/input.fasta \
	--db ../UniProtHymDB.dmnd \
	--outfmt 6 qseqid staxids bitscore qseqid sseqid pident length mismatch gapopen qstart qend sstart send evalue bitscore \
	--sensitive \
	--max-target-seqs 1 \
	--evalue 1e-25 \
	--threads 32 \
	> Diamond.out
	
# download and unpack taxonomy NCBI taxonomy files from https://ftp.ncbi.nlm.nih.gov/pub/taxonomy/new_taxdump/
blobtools add \
	--hits BLAST.out \
	--hits Diamond.out \
	--taxrule bestsumorder \
	--taxdump /path/to/taxdump_files/new_taxdump \
	Dpini_blobtools_out

# Adding coverage information (mapping reads to assembly)
minimap2 -ax map-hifi \
	-t 24 \
	/path/to/assembly/input.fasta \
	/path/to/PacBio/reads/hifi_reads.fastq.gz \
	| samtools view --threads 24 -bS | samtools sort --threads 24 -o Map_HiFi_to_Dpini.bam
blobtools add \
	--cov Map_HiFi_to_Dpini.bam \
	Dpini_blobtools_out
	
# Adding BUSCO data 
blobtools add \
	--busco /path/to/busco/output/BUSCO_full_table.tsv \
	Dpini_blobtools_out

# Blobtool plots
blobtools view --plot --out Blobtool_figures/ --view blob Dpini_blobtools_out
blobtools view --plot --out Blobtool_figures/ --view snail Dpini_blobtools_out
blobtools view --plot --out Blobtool_figures/ --view cumulative Dpini_blobtools_out

### TE identification (D. similis cds accession GCF_021155765.1)

EDTA.pl --genome /path/to/assembly/input.fasta \
	--cds /path/to/GCF_021155765.1_iyDipSimi1.1_cds_from_genomic.fna \
	--overwrite 1 \
	--sensitive 1 \
	--anno 1 \
	--threads 32 \
	--force 1


### Repeat masking

RepeatMasker -lib /path/to/edta_output.mod.EDTA.TElib.fa \
	-pa $SLURM_CPUS_PER_TASK \
	-a \
	-s \
	-no_is \
	-inv \
	-dir ./RepMask \
	-norna \
	-xsmall \
	-nolow \
	/path/to/assembly/input.fasta


### Plot repeat landscape

# get the divsum file (from RepeatMasker output genome.align)
calcDivergenceFromAlign.pl -s genome.divsum /path/to/genome.align

# extract Kimura distance table from divsum file (last 72 lines)
tail -n 72 /path/to/genome.divsum > genome.kimuradist

# plot in R with 'plot_repeat_landscape.R'


### Identifying telomeric repeats

# Search scaffolds for the motif AACCT
tidk search -s AACCT -o dpini -d ./dpini_tidk /path/to/assembly/input.fasta

# plot the output in svg format
tidk plot -t ./dpini_tidk/*.tsv -o Dpini_tidk_plot



### Protein annotation

# BRAKER2 with protein data from Arthropoda OrthoDB11
braker.pl --species=Diprion_pini \
	--genome=/path/to/RepeatMasker/output/input_assembly.fasta.masked \
	--threads=$SLURM_CPUS_PER_TASK \
	--prot_seq=/path/to/Arthropoda_OrthoDB11.fa \
	--softmasking \
	--useexisting
		
# BRAKER2 with RNA-seq data (D. pini transcriptome) from Peters et al. 2017

# calculate indexes. Only needs to be done once.
STAR --runMode genomeGenerate \
	--runThreadN 24 \
	--genomeDir ../ \
	--genomeFastaFiles ../Dpini_HiFi_PB_ONT_primary_purged_scaffII.fasta.masked \
	--genomeSAindexNbases 12
	
# Run the mapping (NOTE: reads cannot be gzipped)
STAR --genomeDir ../ \
	--readFilesIn /path/to/SRR1503030_1.fastq /path/to/SRR1503030_2.fastq \
	--runThreadN 24 \
	--outFileNamePrefix SRR1503030_RNA_mapped_ \
	--outSAMtype BAM SortedByCoordinate
	
braker.pl --species=Diprion_pini \
	--genome=/path/to/RepeatMasker/output/input_assembly.fasta.masked \
	--threads=$SLURM_CPUS_PER_TASK \
	--bam=/path/to/RNA/mapped/to/genome/SRR1503030_RNA_mapped.bam \
	--softmasking \
	--useexisting

# Combining both BRAKER runs with TSEBRA
tsebra.py 	-c tsebra.cfg \
	-g ../OrthoDB/braker/Augustus/augustus.hints.gtf,../RNA/braker/Augustus/augustus.hints.gtf \
	-e ../OrthoDB/braker/hintsfile.gff,../RNA/braker/hintsfile.gff \
	-o braker_prot+rna_out_tsebra.gtf
gffread -w braker_prot+rna_out_tsebra.fa -g /path/to/input_assembly.fasta.masked braker_prot+rna_out_tsebra.gtf
gffread -y braker_prot+rna_out_tsebra.faa -g /path/to/input_assembly.fasta.masked braker_prot+rna_out_tsebra.gtf
